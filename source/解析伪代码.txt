  │   prIsArp = 1 arp知道ip地址 rarp知道mac地址 承认arp rarp包要被广播

  │   prIsIpv4    = 1 知道是ipv4的包 方便switchx 查dscp优先级
  │   prIpDscp    = PacketByte[offset+1][7:2]     ← DSCP 优先级

  │   prIpDa      = PacketByte[offset+16..+19]     ← 目的 IP

  ├─ 0x86dd → IPv6

  │   prIpDscp    = PacketByte[offset][3:0] + [offset+1][7:6]

  │   prIpDa      = PacketByte[offset+24..+39]    ← 128位 目的 IP
